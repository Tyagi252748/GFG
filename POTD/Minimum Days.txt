class Solution{
public:	
	// calculate the maximum sum with out adjacent
	int findMaxSum(int *arr, int n) {
	   vector<int> dp(n,0);
	   dp[0] = arr[0];
	   for(int i = 1; i < n; i++){
	       int incl = dp[i-2] + arr[i];
	       int excl = dp[i-1] + 0;
	       dp[i] = max(incl,excl);
	   }
	   
	   return dp[n-1];
	}
};
