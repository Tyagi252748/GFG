class Solution
{
    public:
    int countSubstring(string S)
    {
        int ans = 0;
        int n = S.length();
        for(int i = 0; i < n; i++){
            int count = 0;
            for(int j = i; j < n; j++){
                if(S[j] >= 'a' && S[j] <= 'z'){
                    count++;
                }else{
                    count--;
                }
                if(count == 0)ans += 1;
            }
        }
        return ans;
    }
};
