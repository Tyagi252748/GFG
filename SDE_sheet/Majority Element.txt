class Solution{
  public:
     // Function to find majority element in the array
    // a: input array
    // size: size of input array
    int majorityElement(int a[], int size)
    {
        vector<int> freq(1e6 + 2, 0);
        for(int i = 0; i < size; i++){
            freq[a[i]]++;
            if(freq[a[i]] > size/2){
                return a[i];
                break;
            }
        }
        return -1;
    }
};
